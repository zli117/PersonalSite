{{- /* Resolve ./foo.md and foo.md links to the rendered page URL. Fragments (#section) are preserved. External URLs pass through. */ -}} {{- $dest := .Destination -}} {{- $url := $dest -}} {{- $anchor := "" -}} {{- if strings.Contains $dest "#" -}} {{- $parts := split $dest "#" -}} {{- $url = index $parts 0 -}} {{- $anchor = printf "#%s" (index $parts 1) -}} {{- end -}} {{- if and $url (or (strings.HasSuffix $url ".md") (strings.HasSuffix $url ".markdown")) -}} {{- with .Page.GetPage $url -}} {{- $dest = printf "%s%s" .RelPermalink $anchor -}} {{- else -}} {{- warnf "render-link: page %q not found (from %s) — leaving link as-is" $url $.Page.File.Filename -}} {{- end -}} {{- end -}} {{ .Text | safeHTML }}
