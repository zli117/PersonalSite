{{- /* Chapter navigation for a series. Renders nothing unless the current page's CurrentSection has `series: true` in its _index.md. Chapters are ordered by `weight`. */ -}} {{- if .CurrentSection.Params.series -}} {{- $chapters := .CurrentSection.RegularPages.ByWeight -}} {{- $total := len $chapters -}} {{- $curIdx := -1 -}} {{- range $i, $p := $chapters -}} {{- if eq $p.RelPermalink $.RelPermalink -}}{{- $curIdx = $i -}}{{- end -}} {{- end -}} {{- if ge $curIdx 0 -}}
{{- if gt $curIdx 0 -}} {{- $prev := index $chapters (sub $curIdx 1) -}} ← Previous {{ $prev.Title }} {{- else -}} {{- end -}} {{- if lt (add $curIdx 1) $total -}} {{- $next := index $chapters (add $curIdx 1) -}} Next → {{ $next.Title }} {{- else -}} {{- end -}}
{{- end -}} {{- end -}}
